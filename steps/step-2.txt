// перевіряємо наявність параметру запиту 'remove_worker' що вказує на те що запитується видалення
if (isset($_GET['remove_worker'])) {
    // отримуємо з параметру ідентифікатор робітника якого треба видалити
    $id = $_GET['remove_worker'];
    // виконуємо запит до бази данних на видалення робітника за його ідентифікатором
    mysqli_query($db, "DELETE FROM workers WHERE id=$id");
    // перенаправлення на головну сторінку
    header('location: index.php');
}
